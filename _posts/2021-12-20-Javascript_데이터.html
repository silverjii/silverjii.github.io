                    const fruits = ['Apple', 'Banana', 'Cherry', 'Orange']
                    const toObject(a,b, ...c){
                        return{
                            a : a,
                            b : b,
                            c : c
                        }
                    }
                    console.log(toObject(...fruits)) 
                    // {a: 'Apple', b: 'Banana', c: Array(2)}
                    
                
                    // 함수 축약
                    // step 1
                    function toObject(a,b, ...c){
                        return{
                            a,
                            b,
                            c   // 변수와 속성 이름이 같으면 하나만 적는다.
                        }
                    }

                    // step 2
                    const toObject = (a,b, ...c) => ({a,b,c})
                    console.log(toObject(...fruits)) 
                    // {a: 'Apple', b: 'Banana', c: Array(2)}

                    // 원시 데이터 : String, Number, Boolean, undefined, null  
                    // 원시 데이터 - 데이터 자체가 변하지 않은 성질을 가지고 있음. 생긴 것이 다르면 다른 데이터 
                    // 참조형 데이터 : Objec, Array, Function - 생긴 것이 똑같아도 같은 데이터가 아닐 수도 있다.

                    let a = 1
                    let b = 4
                    console.log(a, b, a ===b) // 1 4 false

                    b = a
                    console.log(a, b, a === b) // 1 1 true

                    a = 7
                    console.log(a, b, a === b) // 7 1 false

                    let c = 1
                    console.log(b, c, b === c ) // 1 1 true
                    
                    let a = {k : 1}
                    let b = {k : 1}
                    console.log(a, b, a === b )  
                    // {k:1} {k:1} false - 참조형 데이터는 새로운 값을 만들 때 마다 새로운 메모리 주소에 할당된다.

                    a.k = 7
                    b = a 
                    console.log(a, b, a === b ) // {k:7} {k:7} true 

                    a.k=2
                    console.log(a,b, a === b) 
                    // {k:2} {k:2} true - 위에 b=a라고 했기에 k:2, 
                    // 같은 메모리를 바라보고 있는 여러 변수들이 있을 때 
                    //한쪽의 변수에 있는 값을 수정하면 의도하지 않게 다른 변수들도 값이 바뀌어져 있다. 

                    let c = b
                    console.log(a, b, c, a === c ) // {k:2} {k:2} {k:2} true

                    a.k = 9
                    console.log(a, b, c, a ===  c ) // {k:9} {k:9} {k:9} true
